#!/bin/bash

while true
do
echo "===== MENU ====="
echo "1. IP Address"
echo "2. Disk Usage"
echo "3. Memory Usage"
echo "4. Exit"

read -p "Enter choice: " choice

case $choice in
1)
hostname -I
;;

2)
df -h
;;

3)
free -h
;;

4)
exit
;;

*)
echo "Invalid Option"
;;
esac
done


pipeline {

    agent any

    environment {
        INSTITUTE_NAME = "Pathnex"
        TEAM = "DevOps"
        ENV = "prod"
        PROJECT = "Pathnex-Training"
    }

    stages {

        stage('Infrastructure') {
            steps {
                sh '''
                terraform init
                terraform apply -auto-approve
                '''
            }
        }

        stage('App Deployment') {
            steps {
                sh '''
                ansible-playbook user.yml
                kubectl apply -f deployment.yaml
                '''
            }
        }

        stage('Monitoring Checks') {
            steps {
                sh '''
                echo "IP Address"
                hostname -I

                echo "Disk Usage"
                df -h

                echo "Memory Usage"
                free -h
                '''
            }
        }
    }
}



stages:
  - infrastructure
  - app_deployment
  - monitoring_checks

variables:
  INSTITUTE_NAME: "Pathnex"
  TEAM: "DevOps"
  ENV: "prod"
  PROJECT: "Pathnex-Training"

  script:

    - terraform apply -auto-approve
