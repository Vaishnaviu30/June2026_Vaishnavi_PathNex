provider "aws" {
  region = "ap-south-1"
}

resource "aws_key_pair" "key" {
  key_name   = "pathnex-key"
  public_key = file("~/.ssh/id_rsa.pub")
}

resource "aws_security_group" "sg" {
  name = "pathnex-sg"

  ingress {
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  ingress {
    from_port   = 80
    to_port     = 80
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port = 0
    to_port   = 0
    protocol  = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

resource "aws_instance" "server" {
  ami                    = "ami-0f58b397bc5c1f2e8"
  instance_type          = "c5.xlarge"
  key_name               = aws_key_pair.key.key_name
  vpc_security_group_ids = [aws_security_group.sg.id]

  tags = {
    Name = "Pathnex-Server"
  }
}

resource "aws_ebs_volume" "volume" {
  availability_zone = aws_instance.server.availability_zone
  size              = 10
}

resource "aws_volume_attachment" "attach" {
  device_name = "/dev/xvdf"
  volume_id   = aws_ebs_volume.volume.id
  instance_id = aws_instance.server.id
}



---
- hosts: web
  become: yes

  tasks:

    - name: Install Nginx
      yum:
        name: nginx
        state: present

    - name: Deploy HTML page
      copy:
        dest: /usr/share/nginx/html/index.html
        content: |
          <h1>Welcome to Pathnex</h1>

    - name: Enable and Start Nginx
      service:
        name: nginx
        state: started
        enabled: yes


apiVersion: apps/v1
kind: Deployment

metadata:
  name: nginx-deployment

spec:
  replicas: 2

  selector:
    matchLabels:
      app: nginx

  template:
    metadata:
      labels:
        app: nginx

    spec:
      containers:
      - name: nginx
        image: nginx

        volumeMounts:
        - mountPath: /usr/share/nginx/html
          name: data

      volumes:
      - name: data
        persistentVolumeClaim:
          claimName: nginx-pvc

apiVersion: v1
kind: PersistentVolumeClaim

metadata:
  name: nginx-pvc

spec:
  accessModes:
    - ReadWriteOnce

  resources:
    requests:
      storage: 1Gi

apiVersion: v1
kind: Service

metadata:
  name: nginx-service

spec:
  type: NodePort

  selector:
    app: nginx

  ports:
  - port: 80
    targetPort: 80
    nodePort: 30080



#!/bin/bash

echo "===== CPU ====="
top -bn1 | grep "Cpu"

echo

echo "===== Memory ====="
free -h

echo

echo "===== Disk ====="
df -h

echo

echo "===== IP ====="
hostname -I

echo

echo "===== Top Process ====="
ps -eo pid,comm,%cpu --sort=-%cpu | head


pipeline {

    agent any

    environment {
        INSTITUTE_NAME = "Pathnex"
        TEAM = "DevOps"
        ENV = "prod"
        PROJECT = "Pathnex-Training"
    }

    stages {

        stage('Infrastructure') {
            steps {
                sh 'terraform init'
                sh 'terraform apply -auto-approve'
            }
        }

        stage('App Deployment') {
            steps {
                sh 'ansible-playbook -i inventory nginx.yml'
                sh 'kubectl apply -f pvc.yaml'
                sh 'kubectl apply -f deployment.yaml'
                sh 'kubectl apply -f service.yaml'
            }
        }

        stage('Monitoring Checks') {
            steps {
                sh 'chmod +x health-check.sh'
                sh './health-check.sh'
            }
        }
    }

    post {
        always {
            echo "Project: ${PROJECT}"
            echo "Environment: ${ENV}"
            echo "Institute: ${INSTITUTE_NAME}"
            echo "Team: ${TEAM}"
        }

        success {
            echo "Pipeline executed successfully."
        }

        failure {
            echo "Pipeline execution failed."
        }
    }
}



stages:
  - infrastructure
  - app_deployment
  - monitoring_checks

variables:
  INSTITUTE_NAME: "Pathnex"
  TEAM: "DevOps"
  ENV: "prod"
  PROJECT: "Pathnex-Training"

  stage: infrastructure
  script:
    - terraform init
    - terraform apply -auto-approve

  stage: app_deployment

    - kubectl apply -f deployment.yaml
    - kubectl apply -f service.yaml

monitoring_checks:
  stage: monitoring_checks
  script:
    - chmod +x health-check.sh
    - ./health-check.sh
